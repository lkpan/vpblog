# 按需引入

需要安装 npm install babel-plugin-component -D

# 配置文件

在 babel.config.js 中配置

"plugins": [
    [
      "component",
      {
        "libraryName": "element-ui",
        "styleLibraryName": "theme-chalk"
      }
    ]
  ]

# 只引入部分

在 mian.js 中写入

import { Button, Select } from 'element-ui';
