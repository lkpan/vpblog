# 插入数据

const mysql = require('mysql')
const db = mysql.createPool({
    host: '127.0.0.1',
    user: 'root',
    password: 'admin123',
    database: 'myday1'
})
检查是否链接
db.query('select 1', (err, results) => {
    if(err) return console.log((err.message));
    console.log(results);
})

查询user表所有数据
const sqlStr = 'select * from users'
db.query(sqlStr, (err, results) => {
    //查询数据失败
    if(err) return console.log(err.message);
    //如果成功
    console.log(results);//返回数组{[......]}
})

插入数据
const user = {username: 'superman', password: 'pdd123', status:0}
//定义待执行的sql语句
const sqlStr = 'insert into users (username, password, status) values (?, ?, ?) '
//执行sql语句
db.query(sqlStr, [user.username, user.password, user.status], (err, result) => {
    if(err) return console.log(err.message);
    //成功
    //此时result是一个对象
    if(result.affectedRows === 1) {
        console.log('插入数据成功');
    }
})


//便捷插入数据
const user = {username: 'superman1', password: 'pdd1231', status:0}
const sqlStr = 'insert into users set ? '
//执行sql语句
db.query(sqlStr, user, (err, results) => {
    if(err) return console.log(err.message);
    if(results.affectedRows === 1) {
        console.log('插入数据成功');
    }

})

# 删除数据

//删除数据
const sqlStr = 'delete from users where id=?'
db.query(sqlStr, 5, (err, results) => {
    if(err) return console.log(err.message);
    if(results.affectedRows === 1) {
        console.log('更新成功');
    }
})
//标记删除,用状态标记即可
const sqlStr = 'update users set status=? id=2'
db.query(sqlStr, [1, 6], (err, results) => {
    if(err) return console.log(err.message);
    if(results.affectedRows === 1) {
        console.log('更新成功');
    }
})

# 身份认证

如果只是前端建议用session，前后端交互使用jwt，可跨域

jsonwebtoken包将用户信息转换成token， express-jwt包  token还原成用户信jwt是一种加密方式

只要配置成功了jwt中间件，就可以把解析出来的用户信息挂载到req。user属性上，新版本user改成了auth
